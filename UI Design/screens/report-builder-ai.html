SELECT channel, category, month,
  SUM(revenue) as revenue,
  AVG(margin) as avg_margin,
  LAG(AVG(margin)) OVER (PARTITION BY channel,
    category ORDER BY month) as prev_margin
FROM orders_analytics
WHERE date >= '2025-11-01'
GROUP BY channel, category, month
ORDER BY month DESC
